# Shimi's Candle Sentiment (user-changeable colors via DefineGlobalColor)

input showpricecolor = yes;
input showzigzag     = yes;

# User-changeable defaults (edit in Global Colors tab)
DefineGlobalColor("Bullish", Color.CYAN);
DefineGlobalColor("Bearish", CreateColor(102, 153, 255));  # Baby Blue

def highclose     = (close - low) >= (3 * (high - low) / 4);
def lowclose      = (close - low) <= ((high - low) / 4);
def YTCbullcandle = close > high[1];
def YTCbearcandle = close < low[1];

def UpBar   = highclose and YTCbullcandle;
def DownBar = lowclose  and YTCbearcandle;

def insidebar  = high < high[1] and low > low[1];
def outsidebar = high >= high[1] and low <= low[1];

# Sentiment state
def x = if UpBar then 1 else if DownBar then 0 else x[1];

# Apply price color
AssignPriceColor(
    if showpricecolor
    then (if x > 0 then GlobalColor("Bullish") else GlobalColor("Bearish"))
    else Color.CURRENT
);

# ZigZag on sentiment flips
def sh  = if x[-1] == 0 and x == 1 then 1 else 0;
def sl  = if x[-1] == 1 and x == 0 then 1 else 0;
def sh1 = if sh then high else Double.NaN;
def sl1 = if sl then low  else Double.NaN;

plot zz = if showzigzag
          then (if sh > sl then sh1 else sl1)
          else Double.NaN;
zz.EnableApproximation();
